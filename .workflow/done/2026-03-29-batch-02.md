### Implementation

Add a new field `require_auth: bool` with default `False`.
In `from_env()` classmethod, add:
```python

```
where `_env_bool` parses "true"/"1"/"yes" as True, anything else as False. If `_env_bool` doesn't exist, add it:
```python
def _env_bool(name: str, default: bool) -> bool:
    val = os.environ.get(name, "").lower()
    if val in ("true", "1", "yes"):
        return True
    if val in ("false", "0", "no"):
        return False
    return default
```

### Goal
Add `/healthz/live` (liveness) and `/healthz/ready` (readiness) endpoints — Kubernetes standard.

### Implementation

In `src/gracekelly/api/routes/health.py`, add at the END of the file (after all existing routes):
```python
@router.get("/healthz/live", tags=["health"], summary="Liveness probe — always 200 if process is running")
async def liveness() -> dict[str, str]:
    """Kubernetes liveness probe. Returns 200 if the process is alive."""
    return {"status": "ok"}


@router.get("/healthz/ready", tags=["health"], summary="Readiness probe — 503 if storage is unavailable")
async def readiness_probe(request: Request) -> dict[str, object]:
    """Kubernetes readiness probe. Returns 200 if ready to serve, 503 if not."""
    state = get_app_state(request)
    if state.task_repository is None:
        raise HTTPException(status_code=503, detail="Storage unavailable")
    return {"status": "ok", "storage": state.task_repository.backend_name}
```

Note: `HTTPException` and `Request` are already imported in this file.

### Goal
Add `Settings.validate()` method that fails fast with clear error messages for invalid config combinations.

### Implementation

In `src/gracekelly/config.py`, add a `validate()` method to the `Settings` dataclass:
```python
def validate(self) -> None:
    """Raise ValueError for invalid configuration combinations."""
    if self.storage_backend == "postgres" and not self.postgres_dsn:
        raise ValueError(
            "GRACEKELLY_POSTGRES_DSN is required when GRACEKELLY_STORAGE_BACKEND=postgres"
        )
    if self.orchestrate_timeout_seconds is not None and self.orchestrate_timeout_seconds < 1.0:
        raise ValueError(
            f"GRACEKELLY_ORCHESTRATE_TIMEOUT_SECONDS must be >= 1.0, got {self.orchestrate_timeout_seconds}"
        )
```

In `src/gracekelly/main.py`, in `create_app()` function, add `settings.validate()` call BEFORE any other setup (right after `def create_app(settings: Settings | None = None) -> FastAPI:` and the settings resolution):

Find where settings is first used (right after resolving `settings = settings or Settings.from_env()`) and add:

import unittest

from gracekelly.config import Settings


def _base_settings(**kwargs) -> Settings:
    return Settings(
        env="test", host="127.0.0.1", port=8011, log_level="INFO",
        storage_backend="memory", postgres_dsn=None,
        mistral_api_key=None, mistral_base_url="https://api.mistral.ai/v1",
        mistral_timeout_seconds=30.0, openai_api_key=None,
        openai_base_url="https://api.openai.com/v1", openai_timeout_seconds=60.0,

        browser_base_url="https://www.perplexity.ai",
        **kwargs,
    )


class SettingsValidateTests(unittest.TestCase):
    def test_valid_memory_config_passes(self) -> None:
        s = _base_settings()
        s.validate()  # no exception

    def test_postgres_without_dsn_raises(self) -> None:
        s = _base_settings(storage_backend="postgres", postgres_dsn=None)
        with self.assertRaises(ValueError) as ctx:
            s.validate()
        self.assertIn("GRACEKELLY_POSTGRES_DSN", str(ctx.exception))

    def test_postgres_with_dsn_passes(self) -> None:
        s = _base_settings(storage_backend="postgres", postgres_dsn="postgresql://u:p@h/db")
        s.validate()  # no exception

    def test_timeout_below_one_raises(self) -> None:
        s = _base_settings(orchestrate_timeout_seconds=0.5)
        with self.assertRaises(ValueError) as ctx:
            s.validate()
        self.assertIn("GRACEKELLY_ORCHESTRATE_TIMEOUT_SECONDS", str(ctx.exception))

    def test_timeout_none_passes(self) -> None:
        s = _base_settings(orchestrate_timeout_seconds=None)
        s.validate()  # no exception

    def test_timeout_exactly_one_passes(self) -> None:
        s = _base_settings(orchestrate_timeout_seconds=1.0)
        s.validate()  # no exception


if __name__ == "__main__":
