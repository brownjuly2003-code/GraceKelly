### Goal
Add `X-Request-ID` header middleware: echo client header or auto-generate UUID.

### Implementation

Add to `src/gracekelly/middleware.py` at the end of the file:
```python
def setup_correlation_id(app: FastAPI) -> None:
    @app.middleware("http")
    async def correlation_id_middleware(
        request: Request, call_next: Callable[[Request], Awaitable[Response]]
    ) -> Response:
        req_id = request.headers.get("x-request-id") or str(uuid.uuid4())
        response = await call_next(request)
        response.headers["X-Request-ID"] = req_id
        return response
```

Add `import uuid` to imports in `middleware.py` (it already imports from `collections`, add `uuid` to stdlib imports).

In `src/gracekelly/main.py`, find the section where `setup_request_metrics(app)` is called and add `setup_correlation_id(app)` right after it. Also add `from gracekelly.middleware import ... setup_correlation_id` to the import.

## Task A4: RFC 7807 Problem Details error responses

**task_id:** A4-error-schema
**type:** modify + create
**files_to_modify:** `src/gracekelly/main.py`
**files_to_create:** `tests/test_error_schema.py`
**files_to_read:** `src/gracekelly/main.py` (full)

### Goal
Standardize all 4xx/5xx responses to RFC 7807 Problem Details format.
Currently `RequestValidationError` returns FastAPI's default 422 format.
After this change: all errors return `{"type": "...", "title": "...", "status": N, "detail": "..."}`.

### Implementation

In `src/gracekelly/main.py`, import and add two exception handlers.

Find the imports section and add:
```python
from fastapi import FastAPI, Request
from fastapi.exceptions import RequestValidationError
from fastapi.responses import JSONResponse
from starlette.exceptions import HTTPException as StarletteHTTPException
```
(check which are already imported to avoid duplicates)

In `create_app()` function, AFTER `app = FastAPI(...)` is created but BEFORE middleware setup, add:

```python
@app.exception_handler(StarletteHTTPException)
async def http_exception_handler(request: Request, exc: StarletteHTTPException) -> JSONResponse:
    return JSONResponse(
        status_code=exc.status_code,
        content={
            "type": f"https://httpstatuses.com/{exc.status_code}",
            "title": exc.detail if isinstance(exc.detail, str) else "HTTP Error",
            "status": exc.status_code,
            "detail": exc.detail if isinstance(exc.detail, str) else str(exc.detail),
        },
    )

@app.exception_handler(RequestValidationError)
async def validation_exception_handler(request: Request, exc: RequestValidationError) -> JSONResponse:
    errors = exc.errors()
    detail = "; ".join(
        f"{'.'.join(str(loc) for loc in e['loc'])}: {e['msg']}"
        for e in errors
    )
    return JSONResponse(
        status_code=422,
        content={
            "type": "https://httpstatuses.com/422",
            "title": "Validation Error",
            "status": 422,
            "detail": detail,
        },
    )
```

from __future__ import annotations

import unittest

from fastapi.testclient import TestClient

from gracekelly.config import Settings
from gracekelly.main import create_app


def _make_settings(**kwargs) -> Settings:
    return Settings(
        env="test", host="127.0.0.1", port=8011, log_level="INFO",
        storage_backend="memory", postgres_dsn=None,
        mistral_api_key=None, mistral_base_url="https://api.mistral.ai/v1",

        browser_base_url="https://www.perplexity.ai",
        **kwargs,
    )


class ErrorSchemaTests(unittest.TestCase):
    def setUp(self) -> None:
        self.client = TestClient(create_app(_make_settings()), raise_server_exceptions=False)

    def test_404_has_problem_details_type(self) -> None:
        resp = self.client.get("/api/v1/tasks/nonexistent-task-id")
        self.assertIn("type", resp.json())
        self.assertIn("status", resp.json())

    def test_422_has_problem_details_format(self) -> None:
        resp = self.client.post("/api/v1/orchestrate", json={})
        self.assertEqual(resp.status_code, 422)
        body = resp.json()
        self.assertIn("type", body)
        self.assertIn("title", body)
        self.assertIn("status", body)
        self.assertIn("detail", body)
        self.assertEqual(body["status"], 422)

    def test_400_has_problem_details_format(self) -> None:
        resp = self.client.post(
            "/api/v1/orchestrate",
            json={"prompt": "hello", "model": "nonexistent-model-xyz"},
        )
        self.assertEqual(resp.status_code, 400)
        body = resp.json()
        self.assertIn("type", body)
        self.assertIn("status", body)

    def test_422_detail_mentions_field(self) -> None:
        resp = self.client.post("/api/v1/orchestrate", json={})
        self.assertIn("prompt", resp.json()["detail"])


if __name__ == "__main__":
    unittest.main()
