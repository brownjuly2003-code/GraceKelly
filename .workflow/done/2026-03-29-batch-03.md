from __future__ import annotations

import unittest

from hypothesis import given, settings
from hypothesis import strategies as st


class SimilarityPropertiesTests(unittest.TestCase):
    @given(st.text(min_size=1, max_size=200), st.text(min_size=1, max_size=200))
    @settings(max_examples=50, deadline=2000)
    def test_similarity_symmetry(self, a: str, b: str) -> None:
        from gracekelly.core.similarity import cosine_similarity_texts
        score_ab = cosine_similarity_texts(a, b)
        score_ba = cosine_similarity_texts(b, a)
        self.assertAlmostEqual(score_ab, score_ba, places=10)

    @given(st.text(min_size=1, max_size=200))
    @settings(max_examples=50, deadline=2000)
    def test_self_similarity_is_one(self, text: str) -> None:
        from gracekelly.core.similarity import cosine_similarity_texts
        score = cosine_similarity_texts(text, text)
        self.assertAlmostEqual(score, 1.0, places=6)

    @given(st.text(min_size=1, max_size=200), st.text(min_size=1, max_size=200))
    @settings(max_examples=50, deadline=2000)
    def test_similarity_in_range(self, a: str, b: str) -> None:
        from gracekelly.core.similarity import cosine_similarity_texts
        score = cosine_similarity_texts(a, b)
        self.assertGreaterEqual(score, 0.0)
        self.assertLessEqual(score, 1.0 + 1e-9)


if __name__ == "__main__":
    unittest.main()

### Test file 2: `tests/test_clustering_properties.py`
```python
from __future__ import annotations

import unittest

from hypothesis import given, settings
from hypothesis import strategies as st


class HacClusterPropertiesTests(unittest.TestCase):
    @given(
        st.lists(st.text(min_size=1, max_size=100), min_size=1, max_size=10, unique=True),
        st.floats(min_value=0.01, max_value=0.99),
    )
    @settings(max_examples=50, deadline=5000)
    def test_cluster_count_in_valid_range(self, texts: list[str], threshold: float) -> None:
        from gracekelly.core.clustering_hac import hac_cluster
        clusters = hac_cluster(texts, threshold)
        self.assertGreaterEqual(len(clusters), 1)
        self.assertLessEqual(len(clusters), len(texts))

    @given(
        st.lists(st.text(min_size=1, max_size=100), min_size=1, max_size=10, unique=True),
        st.floats(min_value=0.01, max_value=0.99),
    )
    @settings(max_examples=50, deadline=5000)
    def test_all_texts_in_some_cluster(self, texts: list[str], threshold: float) -> None:
        from gracekelly.core.clustering_hac import hac_cluster
        clusters = hac_cluster(texts, threshold)
        all_clustered = {t for cluster in clusters for t in cluster}
        for text in texts:
            self.assertIn(text, all_clustered)

    @given(
        st.lists(st.text(min_size=1, max_size=100), min_size=1, max_size=10, unique=True),
        st.floats(min_value=0.01, max_value=0.99),
    )
    @settings(max_examples=50, deadline=5000)
    def test_no_text_in_multiple_clusters(self, texts: list[str], threshold: float) -> None:
        from gracekelly.core.clustering_hac import hac_cluster
        clusters = hac_cluster(texts, threshold)
        seen: set[str] = set()
        for cluster in clusters:
            for t in cluster:
                self.assertNotIn(t, seen, f"'{t}' appeared in multiple clusters")
                seen.add(t)


if __name__ == "__main__":
    unittest.main()
```

### Constraints
- `deadline=5000` and `max_examples=50` to keep test suite fast
- Import inside test methods to isolate failures

### done_when
- `python -m pytest tests/test_consensus_properties.py tests/test_clustering_properties.py -v` → all pass
- No crashes on random inputs (hypothesis shrinks failures)

## Task C4: Docker multi-stage build

**task_id:** C4-docker-multistage
**type:** modify
**files_to_modify:** `Dockerfile`
**files_to_read:** `Dockerfile` (full)

### Goal
Use multi-stage Docker build to reduce image size by separating build dependencies from runtime.

### Implementation

Replace existing `Dockerfile` with two-stage build:

```dockerfile
# Stage 1: builder
FROM python:3.12-slim AS builder
WORKDIR /app

# Install build dependencies
RUN pip install --upgrade pip wheel

# Copy and install project
COPY pyproject.toml ./
COPY src/ ./src/
RUN pip install --no-cache-dir --target /install .

# Stage 2: runtime
FROM python:3.12-slim

# Non-root user
RUN groupadd -r gracekelly && useradd -r -g gracekelly gracekelly

WORKDIR /app

# Copy installed packages from builder
COPY --from=builder /install /usr/local/lib/python3.12/dist-packages/
COPY --from=builder /app/src ./src/

# Health check
HEALTHCHECK --interval=30s --timeout=5s --start-period=10s --retries=3 \
  CMD python -c "import urllib.request; urllib.request.urlopen('http://localhost:8011/health')"

USER gracekelly

EXPOSE 8011

CMD ["python", "-m", "uvicorn", "gracekelly.main:create_app", "--factory", \
     "--host", "0.0.0.0", "--port", "8011"]
```

### Constraints
- Must still pass `docker build -t gracekelly-test .` (or verify manually)
- Non-root user must be preserved
- HEALTHCHECK must be preserved

### done_when
- `Dockerfile` has FROM ... AS builder stage
- Build completes without error (verified by: `docker build -t gracekelly-test . --no-cache` if Docker available, else skip Docker verification but confirm Dockerfile syntax is correct)
